
cmake_minimum_required(VERSION 3.6.0)
include(${CMAKE_CURRENT_LIST_DIR}/android-jni-support.cmake)

set(LIB_FILE ${CMAKE_CURRENT_LIST_DIR}/../libs/android-jni-support-static/${CMAKE_BUILD_TYPE}/${ANDROID_ABI}/libandroid-jni-support-static.a)
set(LIB_NAME "android-jni-support-static")

set(UUID_LIB_FILE ${CMAKE_CURRENT_LIST_DIR}/../libs/uuid/${CMAKE_BUILD_TYPE}/${ANDROID_ABI}/libuuid.a)
set(UUID_LIB_NAME "uuid")

LinkExternalStaticLibrary(${LIB_NAME} ${LIB_FILE})
LinkExternalStaticLibrary(${UUID_LIB_NAME} ${UUID_LIB_FILE})

add_library(android-jni-support-lib INTERFACE)
add_library(androidjnisupport::lib  ALIAS android-jni-support-lib)

target_link_libraries(android-jni-support-lib INTERFACE android-jni-support-static uuid)

target_include_directories(
  android-jni-support-lib
  INTERFACE
  ${CMAKE_CURRENT_LIST_DIR}/../src/main/cpp/includes/public
)

target_compile_options(
  android-jni-support-lib
  INTERFACE
  ${SHAREDLIBS_CFLAGS}
)

set_target_properties(
  android-jni-support-lib
  PROPERTIES
  INTERFACE_INCLUDE_DIRECTORIES 
  ${CMAKE_CURRENT_LIST_DIR}/../src/main/cpp/includes/public
)
